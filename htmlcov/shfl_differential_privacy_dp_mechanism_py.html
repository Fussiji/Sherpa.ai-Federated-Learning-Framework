8class RandomizedResponseCoins(DataAccessDefinition): 

9 """ 

10 This class uses a simple mechanism to add randomness for binary data. This algorithm is described 

11 by Cynthia Dwork and Aaron Roth in "The algorithmic Foundations of Differential Privacy". 

12 

13 1.- Flip a coin 

14 

15 2.- If tails, then respond truthfully. 

16 

17 3.- If heads, then flip a second coin and respond "Yes" if heads and "No" if tails. 

18 

19 Input data must be binary, otherwise exception will be raised. 

20 

21 # Arguments 

22 prob_head_first: float in [0,1] representing probability to use a random response instead of true value. 

23 This is equivalent to prob_head of the first coin flip algorithm described by Dwork. 

24 prob_head_second: float in [0,1] representing probability of respond true when random answer is provided. 

25 Equivalent to prob_head in the second coin flip in the algorithm. 

26 

27 # References 

28 - [The algorithmic foundations of differential privacy]( 

29 https://www.cis.upenn.edu/~aaroth/Papers/privacybook.pdf) 

30 """ 

31 def __init__(self, prob_head_first=0.5, prob_head_second=0.5): 

32 self._prob_head_first = prob_head_first 

33 self._prob_head_second = prob_head_second 

34 

35 def apply(self, data): 

36 """ 

37 Implements the two coin flip algorithm described by Dwork. 

38 """ 

39 _check_binary_data(data) 

40 size = _get_data_size(data) 

41 

42 first_coin_flip = np.random.rand(size) > self._prob_head_first 

43 second_coin_flip = np.random.rand(size) < self._prob_head_second 

44 

45 result = data * first_coin_flip + (1 - first_coin_flip) * second_coin_flip 

46 

47 if np.isscalar(data): 

48 result = int(result) 

49 

50 return result

53class RandomizedResponseBinary(DataAccessDefinition): 

54 """ 

55 Implements the most general binary randomized response algorithm. Both the input and output are binary 

56 arrays or scalars. The algorithm is defined through the conditional probabilities 

57 

58 - P( output=0 | input=0 ) = f0 

59 - P( output=1 | input=1) = f1 

60 

61 For f0=f1=0 or 1, the algorithm is not random. It is maximally random for f0=f1=1/2. 

62 This class contains, for special cases of f0, f1, the class RandomizedResponseCoins. 

63 

64 Input data must be binary, otherwise exception will be raised. 

65 

66 # Arguments 

67 f0: float in [0,1] representing the probability of getting 0 when the input is 0 

68 f1: float in [0,1] representing the probability of getting 1 when the input is 1 

69 """ 

70 

71 def __init__(self, f0, f1): 

72 self._f0 = f0 

73 self._f1 = f1 

74 

75 def apply(self, data): 

76 """ 

77 Implements the general binary randomized response algorithm. 

78 

79 Both the input and output of the method are binary arrays. 

80 """ 

81 _check_binary_data(data) 

82 size = _get_data_size(data) 

83 

84 if size > 1: 

85 # Binary array case 

86 x_response = np.zeros(size) 

87 x_zero = data == 0 

88 x_response[x_zero] = scipy.stats.bernoulli.rvs(1 - self._f0, size=sum(x_zero)) 

89 x_response[~x_zero] = scipy.stats.bernoulli.rvs(self._f1, size=len(data)-sum(x_zero)) 

90 else: 

91 # Scalar case 

92 if data == 0: 

93 x_response = int(scipy.stats.bernoulli.rvs(1 - self._f0, size=1)) 

94 else: 

95 x_response = int(scipy.stats.bernoulli.rvs(self._f1, size=1)) 

96 

97 return x_response

100class LaplaceMechanism(DataAccessDefinition): 

101 """ 

102 Implements the Laplace mechanism for differential privacy defined by Dwork in 

103 "The algorithmic Foundations of Differential Privacy". 

104 

105 Notice that the Laplace mechanism is a randomization algorithm that depends on the sensitivity, 

106 which can be regarded as a numeric query. One can show that this mechanism is 

107 epsilon-differentially private with epsilon = sensitivity/b where b is a constant. 

108 

109 In order to apply this mechanism for a particular value of epsilon, we need to compute 

110 the sensitivity, which might be hard to compute in practice. The framework provides 

111 a method to estimate the sensitivity of a query that maps the private data in a normed space 

112 (see: [SensitivitySampler](../sensitivity_sampler)) 

113 

114 # Arguments: 

115 sensitivity: float representing sensitivity of the applied query 

116 epsilon: float for the epsilon you want to apply 

117 query: Function to apply over private data (see: [Query](../../private/query)). This parameter is optional and \ 

118 the identity function (see: [IdentityFunction](../../private/query/#identityfunction-class)) will be used \ 

119 if it is not provided. 

120 

121 # References 

122 - [The algorithmic foundations of differential privacy]( 

123 https://www.cis.upenn.edu/~aaroth/Papers/privacybook.pdf) 

124 """ 

125 def __init__(self, sensitivity, epsilon, query=None): 

126 if query is None: 

127 query = IdentityFunction() 

128 

129 self._sensitivity = sensitivity 

130 self._epsilon = epsilon 

131 self._query = query 

132 

133 def apply(self, data): 

134 query_result = self._query.get(data) 

135 size = _get_data_size(query_result) 

136 b = self._sensitivity/self._epsilon 

137 

138 return data + np.random.laplace(loc=0.0, scale=b, size=size)

141class ExponentialMechanism(DataAccessDefinition): 

142 """ 

143 Implements the exponential mechanism differential privacy defined by Dwork in  

144 "The algorithmic Foundations of Differential Privacy". 

145  

146 # Arguments: 

147 u: utility function with arguments x and r. It should be vectorized, so that for a \ 

148 particular database x, it returns as many values as given in r. 

149 r: array for the response space. 

150 delta_u: float for the sensitivity of the utility function. 

151 epsilon: float for the epsilon you want to apply. 

152 size: integer for the number of queries to perform at once. If not given it defaults to one. 

153  

154 # References 

155 - [The algorithmic foundations of differential privacy]( 

156 https://www.cis.upenn.edu/~aaroth/Papers/privacybook.pdf) 

157 """ 

158 def __init__(self, u, r, delta_u, epsilon, size=1): 

159 self._u = u 

160 self._r = r 

161 self._delta_u = delta_u 

162 self._epsilon = epsilon 

163 self._size = size 

164 

165 def apply(self, data): 

166 r_range = self._r 

167 u_points = self._u(data, r_range) 

168 p = np.exp(self._epsilon * u_points / (2 * self._delta_u)) 

169 p /= p.sum() 

170 sample = np.random.choice(a=r_range, size=self._size, replace=True, p=p) 

171 

172 return sample
